% LivRev superscript and subscript macros
\def\getfirst#1#2STOP{#1}%
\DeclareRobustCommand*\super[1]{%
\edef\tmpx{\getfirst#1STOP}%
\def\tmpy{-}%
\ifx\tmpx\tmpy
\textsuperscript{-#1}%
\else
\textsuperscript{#1}%
\fi
}%
\DeclareRobustCommand*\sub[1]{%
\edef\tmpx{\getfirst#1STOP}%
\def\tmpy{-}%
\ifx\tmpx\tmpy
\@sub{\selectfont-#1}%
\else
\@sub{\selectfont#1}%
\fi
}%
\def\@sub#1{{\m@th\ensuremath{_{\mbox{\fontsize\sf@size\z@#1}}}}}
